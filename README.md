## Usage

QuickStart :

* Requirements : **haxe** 3.2+ <http://build.haxe.org>

```haxe
import js.Browser.document;

class Main{
	static function main() {
		var ec = js.Echarts.init( document.getElementById("ec") );

		ec.setOption( {
			title: {
				text: "Echarts 入门示例",
			},

			tooltip: {},

			legend: {
				data: ["销量"]
			},

			xAxis: {
				data: ["衬衫","羊毛衫","雪纺衫","裤子","高跟鞋","袜子"]
			},

			yAxis: {},

			series: {
				name: "销量",
				type: "bar",
				data: [5, 20, 36, 10, 10, 20]
			}
		});
	}
}
```
